// OceanJet PortDesk optional Google Sheets endpoint template
function doPost(e){
  const payload = JSON.parse(e.postData.contents || '{}');
  return ContentService.createTextOutput(JSON.stringify({ok:true,receivedAt:new Date().toISOString(),type:payload.type||'backup'})).setMimeType(ContentService.MimeType.JSON);
}
function doGet(){
  return ContentService.createTextOutput(JSON.stringify({ok:true,service:'OceanJet PortDesk Sync'})).setMimeType(ContentService.MimeType.JSON);
}
